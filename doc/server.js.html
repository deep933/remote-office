const express = require("express")
const app = express()
const server = require("http").createServer(app);
const io = require("socket.io")(server)
const path = require('path');
const port = 8080
const { ExpressPeerServer } = require('peer');
const peerServer =  ExpressPeerServer(server, {
    path: '/myapp'
  });

app.use('/peerjs',peerServer)


app.use(express.static(path.join(__dirname, '/build')));
    
io.on('connection', socket => { 

  socket.on("join-room",(roomId,userId)=>{

    socket.join(roomId)
    var room = io.sockets.adapter.rooms[roomId];
    socket.to(roomId).emit("newUserConnect",room.length);

    socket.on("video-offer",(roomId,data)=>{
      console.log("video-offer",data)
      socket.join(roomId)
      socket.to(roomId).broadcast.emit("oncall",data)
    })

    socket.on("video-answer",(roomId,data)=>{
      console.log("video-answer",data)
      socket.join(roomId)
      socket.to(roomId).broadcast.emit("oncall",data)
    })

    socket.on("new-candidate",(roomId,data)=>{
      console.log("new-candidate",data)
      socket.join(roomId)
      socket.to(roomId).broadcast.emit("onnewcandidate",data)
    })

    socket.on('disconnect',()=>{
    })

  })




   

      

      

    });
